101void TimerableService::dispatch(uint32_t timeout) {
102 auto now = (uint64_t) std::chrono::time_point_cast<std::chrono::seconds>(std::chrono::system_clock::now()).time_since_epoch().count();
103 auto index = (now - startTimestamp) % 60;
104 auto &table = timeoutTable[index];
105
106 for (auto iterator = table.begin(); iterator != table.end();) {
107 if ((*iterator)->exceptTimestamp <= now) {
108 onTimeout(*iterator);
109 timeoutMap.erase((*iterator)->fd);
110 delete (*iterator);
111 table.erase(iterator++);
112 } else {
113 iterator++;
114 }
115 }
116
117 event::EventLoop::dispatch(timeout);
118}
