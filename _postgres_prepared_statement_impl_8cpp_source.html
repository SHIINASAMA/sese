150bool sese::db::impl::PostgresPreparedStatementImpl::getColumnType(uint32_t index, sese::db::MetadataType &type) noexcept {
151 const auto META = PQdescribePrepared(conn, stmtName.c_str());
152 if (META == nullptr || PQnfields(META) <= static_cast<int>(index)) {
153 return false;
154 }
155 switch (PQftype(META, static_cast<int>(index))) {
156 case BOOLOID:
157 type = MetadataType::BOOLEAN;
158 break;
159 case TEXTOID:
160 case CHAROID:
161 case VARCHAROID:
162 type = MetadataType::TEXT;
163 break;
164 case INT8OID:
165 type = MetadataType::LONG;
166 break;
167 case INT4OID:
168 type = MetadataType::INTEGER;
169 break;
170 case INT2OID:
171 type = MetadataType::SHORT;
172 break;
173 case FLOAT8OID:
174 type = MetadataType::FLOAT;
175 break;
176 case FLOAT4OID:
177 type = MetadataType::DOUBLE;
178 break;
179 case DATEOID:
180 type = MetadataType::DATE;
181 break;
182 case TIMEOID:
183 type = MetadataType::TIME;
184 break;
185 case TIMESTAMPTZOID:
186 case TIMESTAMPOID:
187 type = MetadataType::DATE_TIME;
188 break;
189 default:
190 type = MetadataType::UNKNOWN;
191 break;
192 }
193 return true;
194}
