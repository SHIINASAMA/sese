36bool sese::db::impl::MariaPreparedStatementImpl::mallocBindStruct(MYSQL_RES *meta, MYSQL_BIND **bind) noexcept {
37 *bind = static_cast<MYSQL_BIND *>(malloc(sizeof(MYSQL_BIND) * meta->field_count));
38 memset(*bind, 0, sizeof(MYSQL_BIND) * meta->field_count);
39 for (unsigned int i = 0; i < meta->field_count; i++) {
40 auto &&item = (*bind)[i];
41 item.is_null = static_cast<decltype(item.is_null)>(malloc(sizeof(char)));
42 *item.is_null = 0;
43
44 // 非定长数据
45 switch (meta->fields[i].type) {
46 case MYSQL_TYPE_BLOB:
47 case MYSQL_TYPE_TINY_BLOB:
48 case MYSQL_TYPE_STRING:
49 case MYSQL_TYPE_VARCHAR:
50 case MYSQL_TYPE_VAR_STRING:
51 item.buffer = malloc(meta->fields[i].length);
52 item.buffer_length = meta->fields[i].length;
53 break;
54 case MYSQL_TYPE_TINY:
55 item.buffer = malloc(1);
56 break;
57 case MYSQL_TYPE_SHORT:
58 item.buffer = malloc(2);
59 break;
60 case MYSQL_TYPE_LONG:
61 item.buffer = malloc(4);
62 break;
63 case MYSQL_TYPE_LONGLONG:
64 item.buffer = malloc(8);
65 break;
66 case MYSQL_TYPE_FLOAT:
67 item.buffer = malloc(4);
68 break;
69 case MYSQL_TYPE_DOUBLE:
70 item.buffer = malloc(8);
71 break;
72 case MYSQL_TYPE_TIME:
73 case MYSQL_TYPE_DATE:
74 case MYSQL_TYPE_DATETIME: {
75 item.buffer = malloc(sizeof(MYSQL_TIME));
76 memset(item.buffer, 0, sizeof(MYSQL_TIME));
77 break;
78 }
79 default:
80 return false;
81 }
82 item.buffer_type = meta->fields[i].type;
83 }
84 return true;
85}

95sese::db::ResultSet::Ptr sese::db::impl::MariaPreparedStatementImpl::executeQuery() noexcept {
96 if (!meta) return nullptr;
97
98 if (mysql_stmt_bind_param(stmt, this->param)) {
99 return nullptr;
100 }
101 MYSQL_BIND *result;
102 if (!mallocBindStruct(meta, &result)) {
103 goto freeResult;
104 }
105 if (mysql_stmt_bind_result(stmt, result)) {
106 goto freeResult;
107 }
108 if (mysql_stmt_execute(stmt)) {
109 goto freeResult;
110 }
111 if (mysql_stmt_store_result(stmt)) {
112 goto freeResult;
113 }
114 return std::make_unique<impl::MariaStmtResultSet>(stmt, result, meta->field_count);
115
116freeResult:
117 freeBindStruct(result, meta->field_count);
118 return nullptr;
119}
