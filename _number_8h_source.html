39 static void toString(StringBuilder &builder, T number, int radix, bool upper_case) noexcept {
40 if (number < 0) {
41 builder.append('-');
42 number = -number;
43 }
44 int count = 0;
45 constexpr auto DIGIT_UP = "0123456789ABCDEF";
46 constexpr auto DIGIT_DOWN = "0123456789abcdef";
47 const auto DIGIT = upper_case ? DIGIT_UP : DIGIT_DOWN;
48 do {
49 int index = number % radix;
50 builder.append(DIGIT[index]);
51 number /= radix;
52 count += 1;
53 } while (number > 0);
54
55 // 直接反转 builder 中部分内存
56 std::reverse(static_cast<char *>(builder.buf()) + builder.length() - count, static_cast<char *>(builder.buf()) + builder.length());
57 }

59 template<typename T>
60 static void toString(StringBuilder &builder, T number, uint16_t placeholder) noexcept {
61 auto int_part = static_cast<int64_t>(number);
62 double frac = number - int_part;
63 double rounding_factor = std::pow(10, placeholder);
64 auto rounded_part = static_cast<int64_t>(frac * rounding_factor);
65
66 if (static_cast<double>(rounded_part) >= rounding_factor) {
67 int_part += 1;
68 rounded_part -= static_cast<int64_t>(rounding_factor);
69 }
70
71 toString(builder, int_part, 10, true);
72 if (placeholder > 0) {
73 builder.append('.');
74 if (rounded_part < 0) {
75 rounded_part = -rounded_part;
76 }
77 toString(builder, rounded_part, 10, true);
78 auto len = number2StringLength(rounded_part, 10);
79 if (len < placeholder) {
80 builder.append(std::string(placeholder - len, '0'));
