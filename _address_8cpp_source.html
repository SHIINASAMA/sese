16bool Address::lookUp(std::vector<Address::Ptr> &result, const std::string &host, int family, int type, int protocol) {
17 struct addrinfo *res, hints{0};
18 hints.ai_flags = 0;
19 hints.ai_family = family;
20 hints.ai_socktype = type;
21 hints.ai_protocol = protocol;
22 hints.ai_addrlen = 0;
23 hints.ai_canonname = nullptr;
24 hints.ai_addr = nullptr;
25 hints.ai_next = nullptr;
26
27 int err = getaddrinfo(host.c_str(), nullptr, &hints, &res);
28 if (err) {
29 return false;
30 }
31
32 auto next = res;
33 while (next) {
34 result.emplace_back(create(next->ai_addr, static_cast<socklen_t>(next->ai_addrlen)));
35 next = next->ai_next;
36 }
37
38 freeaddrinfo(res);
39 return true;
40}
