34int64_t BufferedStream::read(void *buf, size_t length) {
35 /*
36 * 如果读取所需字节数需要缓存两次以下，
37 * 需要进行预读操作，避免频繁的IO操作；
38 * 如果读取所需字节数需要缓存两次及两次以上，
39 * 则处理原先的缓存后，直接操作裸流，减少拷贝次数
40 */
41 if (length <= this->cap) {
42 if (this->len - this->pos >= length) {
43 // 字节数足够 - 不需要预读取
44 memcpy(buf, static_cast<char *>(this->buffer) + this->pos, length);
45 pos += length;
46 return static_cast<int64_t>(length);
47 } else {
48 // 字节数不足 - 需要预读取
49 size_t total = this->len - this->pos;
50 memcpy(buf, static_cast<char *>(this->buffer) + this->pos, total);
51 pos += total;
52 if (0 != preRead()) {
53 if (this->len - this->pos >= length - total) {
54 // 字节数足够
55 memcpy(static_cast<char *>(buf) + total, this->buffer, length - total);
56 pos = length - total;
57 total = length;
58 } else {
59 // 字节数不足，且无法继续读取
60 memcpy(static_cast<char *>(buf) + total, this->buffer, this->len - this->pos);
61 pos = this->len - this->pos;
62 total += this->len - this->pos;
63 }
64 }
65 return static_cast<int64_t>(total);
66 }
67 } else {
68 // 先处理已有缓存
69 size_t total = this->len - this->pos;
70 memcpy(buf, this->buffer, total);
71 this->len = 0;
72 this->pos = 0;
73 // 操作裸流
74 while (true) {
75 size_t read = source->read(static_cast<char *>(buf) + total, (length - total) >= 1024 ? 1024 : length - total);
76 total += static_cast<int64_t>(read);
77 // 无可再读
78 if (read <= 0) break;
79 // 完成目标
80 if (total == length) break;
81 }
82 return static_cast<int64_t>(total);
83 }
84}

86int64_t BufferedStream::write(const void *buf, size_t length) {
87 /*
88 * 如果写入所需字节数需要刷新缓存两次一下，
89 * 则暂时写入缓存，避免频繁的IO操作；
90 * 如果写入所需字节数需要刷新缓存两次及两次以上，
91 * 则处理原先的缓存后，直接操作裸流，减少拷贝次数
92 */
93 if (length <= this->cap) {
94 if (this->cap - this->len >= length) {
95 // 字节数足够 - 不需要刷新
96 memcpy(static_cast<char *>(this->buffer) + this->len, buf, length);
97 this->len += length;
98 return static_cast<int64_t>(length);
99 } else {
100 // 字节数不足 - 需要刷新
101 size_t expect = len - pos;
102 if (expect == flush()) {
103 memcpy(this->buffer, (char *) buf, length);
104 this->len = length;
105 expect = length;
106 return static_cast<int64_t>(expect);
107 } else {
108 // flush 失败
109 return -1;
110 }
111 }
112 } else {
113 // 直接写入
114 if (this->len != this->pos) {
115 // 缓存区有剩余，需要刷新
116 size_t expect = len - pos;
117 if (expect != flush()) {
118 // flush 失败
119 return -1;
120 }
121 }
122
123 int64_t wrote = 0;
124 while (true) {
125 auto rt = source->write(static_cast<const char *>(buf) + wrote, length - wrote >= cap ? cap : length - wrote);
126 if (rt <= 0) return -1;
127 wrote += rt;
128 if (wrote == length) break;
129 }
130
131 return wrote;
132 }
133}

135int64_t BufferedStream::flush() noexcept {

139 len = 0;
