26std::optional<DateTime> DateTimeParser::parse(const std::string &format, const std::string &datetime) {
27 if (format.empty() || datetime.empty())
28 return std::nullopt;
29
30 const char *p_format = format.c_str();
31 auto input = sese::io::InputBufferWrapper(datetime.c_str(), datetime.length());
32
33 int year = 0;
34 int mon = 0;
35 int day = 0;
36 int hour = 0;
37 int min = 0;
38 int sec = 0;
39 int mill = 0;
40 int micro = 0;
41 int utc = 0;

43#define XX(ch, exp_size, save_to) \
44 case ch: { \
45 char *end; \
46 auto c = count(p_format, exp_size); \
47 if (c >= exp_size) { \
48 char buffer[exp_size + 1]{}; \
49 if (input.read(buffer, exp_size) != exp_size) { \
50 return std::nullopt; \
51 } \
52 save_to = std::strtol(buffer, &end, 10); \
53 if (*end != 0) { \
54 return std::nullopt; \
55 } \
56 p_format += exp_size; \
57 } else { \
58 p_format += c; \
59 } \
60 break; \
61 }
62
63 while (*p_format) {
64 switch (*p_format) {
65 XX('y', 4, year)
66 XX('M', 2, mon)
67 XX('d', 2, day)
68 XX('H', 2, hour)
69 XX('m', 2, min)
70 XX('s', 2, sec)
71 XX('f', 3, mill)
72 XX('r', 3, micro)
73 case 'z': {
74 char *end;
75 char buffer[4]{};
76 if (input.read(buffer, 2) != 2) {
77 break;
78 }
79 if (buffer[0] == '+' || buffer[0] == '-') {
80 if (input.read(&buffer[2], 1) != 1) {
81 break;
82 }
83 }
84 utc = std::strtol(buffer, &end, 10);
85 if (*end != 0) {
86 return std::nullopt;

88 p_format += 1;
89 break;
90 }
91 case '*': {
92 while (true) {
93 char c{};

95 if (l == 0 && p_format + 1 <= format.c_str() + format.length()) {
96 break;
97 } else if (c == ' ') {
98 p_format += 1;
99 break;
100 } else {
101 input.trunc(1);
102 }
103 }
104 break;
105 }
106 case '%': {
107 if (p_format + 1 > format.c_str() + format.length()) {
108 break;
109 } else {
110 p_format += 2;
111 break;
112 }
113 }
114 default:
115 p_format += 1;
116 input.trunc(1);
117 break;
118 }

122 struct tm tm = {};

131 tp += std::chrono::milliseconds(mill) + std::chrono::microseconds(micro);
132 tp -= std::chrono::hours(utc);
133
134 // Convert time_point to microseconds
135 uint64_t timestamp = std::chrono::duration_cast<std::chrono::microseconds>(tp.time_since_epoch()).count();
