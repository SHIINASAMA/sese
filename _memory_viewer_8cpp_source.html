17void MemoryViewer::peer(OutputStream *output, void *position, size_t size, bool is_cap) noexcept {
18 auto *p = (uint8_t *) position;
19 for (auto i = 0; i < size; i++) {
20 char word[3]{0, 0, ' '};
21 word[0] = toChar(*(p + i + 0) / 0x10, is_cap);
22 word[1] = toChar(*(p + i + 0) % 0x10, is_cap);
23 if (i + 1 == size) {
24 output->write(word, 2);
25 } else {
26 output->write(word, 3);
27 }
28 }
