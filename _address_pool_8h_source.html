46std::shared_ptr<ADDRESS> sese::net::AddressPool<ADDRESS>::lookup(const std::string &domain) noexcept {
47 auto iterator = pool.addressMap.find(domain);
48 auto inet = std::is_same<ADDRESS, sese::net::IPv4Address>::value ? AF_INET : AF_INET6;
49 if (iterator == pool.addressMap.end()) {
50 auto address = ADDRESS::lookUpAny(domain, inet, IPPROTO_IP);
51 if (nullptr == address) {
52 // 未命中缓存且查找失败
53 return nullptr;
54 } else {
55 auto ip = dynamicPointerCast<ADDRESS>(address);
56 pool.addressMap[domain] = ip;
57 return ip;
58 }
59 } else {
60 return iterator->second;
