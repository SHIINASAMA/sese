27using sese::system::Semaphore;
28
29Semaphore::Ptr Semaphore::create(std::string name, uint32_t initial_count) {
30 auto sem = sem_open(name.c_str(), O_CREAT | O_RDWR, 0666, initial_count);
31 if (sem == SEM_FAILED) {
32 return nullptr;
33 }
34
35 auto res = new Semaphore();
36 res->semaphore = sem;
37 res->sem_name = std::move(name);
38 return std::unique_ptr<Semaphore>(res);
39}
40
41Semaphore::~Semaphore() {
42 sem_close(semaphore);
43 sem_unlink(sem_name.c_str());
44}
45
46bool Semaphore::lock() {
47 return 0 == sem_wait(semaphore);
48}
49
50bool Semaphore::unlock() {
51 return 0 == sem_post(semaphore);
52}
53
54bool Semaphore::tryLock(std::chrono::milliseconds ms) {
55#ifdef SESE_PLATFORM_LINUX
56 struct timespec time {};
57 time.tv_nsec = 1000000 * ms.count();
58 return 0 == sem_timedwait(semaphore, &time);
59#else
60 return 0 == sem_trywait(semaphore);
61#endif
62}
