18void MD5Util::encode(InputStream *input, OutputStream *output) noexcept {
19 uint32_t result[4]{A, B, C, D};
20 unsigned char buffer[64];
21 uint64_t length = 0;
22
23 bool is_break = false;
24 while (!is_break) {
25 int64_t len = input->read(buffer, 64);
26 if (len == 64) {
27 length += len;
28 } else if (len == 0) {
29 memcpy(&buffer[len], PADDING, 64 - len - 8);
30 length *= 8;
31 length = ToLittleEndian64(length);
32 memcpy(&buffer[56], &length, 8);
34 is_break = true;
35 } else if (len < 56) {
36 length += len;
37 memcpy(&buffer[len], PADDING, 64 - len - 8);
38 length *= 8;
39 length = ToLittleEndian64(length);
40 memcpy(&buffer[56], &length, 8);
41 is_break = true;
42 } else {
43 length += len;
44 memcpy(&buffer[len], PADDING, 64 - len);
45 // 倒数第二次变换
46 MD5Util::transform(result, buffer);
47 memset(buffer, 0, 56);
48 length *= 8;
49 length = ToLittleEndian64(length);
50 memcpy(&buffer[56], &length, 8);
51 is_break = true;
52 }
53
54 MD5Util::transform(result, buffer);
55 }
56
57 output->write(result, 16);
58}

66std::unique_ptr<char[]> MD5Util::encode(InputStream *input, bool is_cap) noexcept {
67 ByteBuilder dest(16);
68 encode(input, &dest);
69 unsigned char buffer[16];
70 auto rt = std::unique_ptr<char[]>(new char[33]); // GCOVR_EXCL_LINE
71 dest.read(buffer, 16);
72 for (size_t i = 0; i < 16; i++) {
73 rt[i * 2 + 1] = MemoryViewer::toChar(buffer[i] % 0x10, is_cap);
74 rt[i * 2 + 0] = MemoryViewer::toChar(buffer[i] / 0x10, is_cap);
75 }
76 rt[32] = 0;
77 return rt;
78}

163uint32_t MD5Util::F(uint32_t x, uint32_t y, uint32_t z) noexcept {
