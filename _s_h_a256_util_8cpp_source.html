352 Bitset32 *p0 = &buffer[0];
353 Bitset32 *p1 = &buffer[1];

373void SHA256Util::encode(Bitset32 *value, Bitset32 *buffer, Bitset32 k) noexcept {
374 Bitset32 choice = (e & f) ^ ((~e) & g);
375 Bitset32 majority = (a & b) ^ (a & c) ^ (b & c);
376 Bitset32 sigma0 = a.rightRotate2() ^ a.rightRotate13() ^ a.rightRotate22();
377 Bitset32 sigma1 = e.rightRotate6() ^ e.rightRotate11() ^ e.rightRotate25();
378 Bitset32 tmp1 = h + sigma1 + choice + k + *buffer;
379 Bitset32 tmp2 = sigma0 + majority;
380
381 h = g;
382 g = f;
383 f = e;
384 e = d + tmp1;
385 d = c;
386 c = b;
387 b = a;
388 a = tmp1 + tmp2;
389}
