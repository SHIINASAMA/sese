42char sstr::getSizeFromUTF8Char(char ch) {
43 if ((ch & 0b10000000) == 0b00000000) {
44 return 1;
45 } else if ((ch & 0b11100000) == 0b11000000) {
46 return 2;
47 } else if ((ch & 0b11110000) == 0b11100000) {
48 return 3;
49 } else if ((ch & 0b11111000) == 0b11110000) {
50 return 4;
51 } else {
52 return -1;
53 }

56char sstr::getUTF8SizeFromUnicodeChar(SChar ch) {
57 if (static_cast<uint32_t>(ch) <= 0x7f) {
58 return 1;
59 } else if (static_cast<uint32_t>(ch) > 0x7f && static_cast<uint32_t>(ch) <= 0x7ff) {
60 return 2;

62 return 3;

64 return 4;
65 } else {
66 return -1;

79SChar sstr::getUnicodeCharFromUTF8Char(char size, const char *ch) {
80 switch (size) {
81 case 1:
82 return SChar(*ch & 0b01111111);
83 case 2:
84 return SChar((*ch & 0b00011111) << 6 | (*(ch + 1) & 0b00111111));
85 case 3:
86 return SChar((*ch & 0b00001111) << 12 | (*(ch + 1) & 0b00111111) << 6 | (*(ch + 2) & 0b00111111));
87 case 4:
88 return SChar((*ch & 0b00000111) << 18 | (*(ch + 1) & 0b00111111) << 12 | (*(ch + 2) & 0b00111111) << 6 | (*(ch + 3) & 0b00111111));
89 default:
90 return null_char;
91 }
92}

99static bool insertUnicodeChar2UTF8String(char *destination, uint32_t code, size_t n) {
100 if (1 == n) {
101 *(destination + 0) = static_cast<char>(0b01111111 & code);
102 } else if (2 == n) {
103 *(destination + 0) = static_cast<char>(0b11000000 | (0b0000011111000000 & code) >> 6);
104 *(destination + 1) = static_cast<char>(0b10000000 | (0b0000000000111111 & code));
105 } else if (3 == n) {
106 *(destination + 0) = static_cast<char>(0b11100000 | (0b1111000000000000 & code) >> 12);
107 *(destination + 1) = static_cast<char>(0b10000000 | (0b0000111111000000 & code) >> 6);
108 *(destination + 2) = static_cast<char>(0b10000000 | (0b0000000000111111 & code));
109 } else if (4 == n) {
110 *(destination + 0) = static_cast<char>(0b11110000 | (0b000111000000000000000000 & code) >> 18);
111 *(destination + 1) = static_cast<char>(0b10000000 | (0b000000111111000000000000 & code) >> 12);
112 *(destination + 2) = static_cast<char>(0b10000000 | (0b000000000000111111000000 & code) >> 6);
113 *(destination + 3) = static_cast<char>(0b10000000 | (0b000000000000000000111111 & code));
114 } else {
115 return false;
116 }
117 return true;
118}

330SString SString::fromUCS2LE(const wchar_t *str) {
331 // on dos-like, wchar_t use 2 bytes
332 // as utf-16le
333 // on *nix, wchar_t use 4 bytes by default
334 // as utf-32le
335 SString s_string;
336 const wchar_t *p = str;
337 while (L'\0' != *p) {
338 s_string._size += getUTF8SizeFromWChat(*p);
339 p++;

514SString SStringView::trim() const {
515 auto new_size = _size;
516 for (auto i = 0; i < _size; i++) {
517 if (_data[i] == ' ') {
518 new_size--;
519 }
520 }
521
522 auto new_cap = (new_size / BLOCK_SIZE + 1) * BLOCK_SIZE;
523
524 char *new_data = static_cast<char *>(malloc(new_cap));
525 char *p = _data;
526 while (*p == ' ') p++;
527 memcpy(new_data, p, new_size);
528 new_data[new_size] = '\0';
529
530 SString string;
531 string._size = new_size;
532 string._capacity = new_cap;
533 string._data = new_data;
534 return string;
535}

581std::vector<SString> SStringView::split(const char *str) const {
582 std::vector<SString> v;
583 auto size = getByteLengthFromUTF8String(str);
584
585 std::string::size_type pos1, pos2;
586 pos2 = BM(_data, str);
587 pos1 = 0;
588 while (true) {
589 v.emplace_back(_data + pos1, pos2 - pos1);
590
591 pos1 = pos2 + size;
592 pos2 = BM(_data + pos1, str);
593 if (-1 == pos2) {
594 v.emplace_back(SString::fromUTF8(_data + pos1));
595 break;
596 } else {
597 pos2 += pos1;
598 }
599 }
600 return v;
601}

652 std::vector<SChar> chars;
