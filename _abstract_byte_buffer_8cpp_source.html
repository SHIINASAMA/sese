172 while (true) {
173 // 当前单元能提供的剩余读取量
174 // 当前单元能满足读取需求
175 if (const size_t CURRENT_READ_NODE_REMAINING = currentReadNode->length - currentReadPos ;need_read <= CURRENT_READ_NODE_REMAINING) {
176 memcpy(static_cast<char *>(buffer) + actual_read, static_cast<char *>(currentReadNode->buffer) + currentReadPos, need_read);
177 actual_read += static_cast<int64_t>(need_read);
178 currentReadPos += need_read;
179 break;
180 }
181 // 当前单元不能满足读取需求
182 else {
183 memcpy(static_cast<char *>(buffer) + actual_read, static_cast<char *>(currentReadNode->buffer) + currentReadPos, CURRENT_READ_NODE_REMAINING);
184 actual_read += static_cast<int64_t>(CURRENT_READ_NODE_REMAINING);
