29bool AsioHttpClient::init(const std::string &url, const std::string &proxy) {
30 auto url_result = RequestParser::parse(url);
31 req = std::move(url_result.request);
32 if (!proxy.empty()) {
33 auto proxy_result = RequestParser::parse(proxy);
34 if (proxy_result.address == nullptr) {
35 return false;
36 }
37 req->setUrl(url);
38 req->set("via:", proxy);
39 req->set("proxy-connection", "keep-alive");
40 address = std::move(proxy_result.address);
41 ssl = strcmpDoNotCase("https", proxy_result.url.getProtocol().c_str());
42 } else {
43 if (url_result.address == nullptr) {
44 return false;
45 }
46 address = std::move(url_result.address);
47 ssl = strcmpDoNotCase("https", url_result.url.getProtocol().c_str());
48 }
49
50 resp = std::make_unique<Response>();
51 req->set("user-agent", "sese-httpclient/1.0");
52 req->set("connection", "keep-alive");
53 cookies = std::make_shared<CookieMap>();
54 req->setCookies(cookies);
55
56 return true;
57}

59bool AsioHttpClient::request() {
60 // 允许重连一次
61 int times = 0;
62 if (first) {
63 times += 1;
64 } else {
65 first = false;
66 }
67 while (!HttpUtil::sendRequest(this, req.get())) {
68 if (times > 1) {
69 return false;
70 }
71
72 times += 1;
73 asio::ip::tcp::endpoint endpoint(convert(address), address->getPort());
74 code = socket.connect(endpoint, code);
75 if (code) {
76 return false;
77 }
78
79 if (ssl) {
80 code = sslSocket.handshake(asio::ssl::stream_base::client, code);
81 if (code) {
82 return false;
83 }
84 }
85 }
86
87 auto response_status = HttpUtil::recvResponse(this, resp.get());
88 if (!response_status) {
89 return false;
90 }
91
92 const auto CONNECTION_VALUE = resp->get("connection", "close");
93 if (strcmpDoNotCase("close", CONNECTION_VALUE.c_str())) {
94 first = true;
95 if (ssl) {
96 code = sslSocket.shutdown(code);
97 }
98 socket.close();
99 }
100
101 // 自动应用 cookie
102 const auto DEST = req->getCookies();
103 for (auto &&item: *resp->getCookies()) {
104 DEST->add(item.second);
105 }
106
107 return true;
108}
