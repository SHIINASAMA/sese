40int sese::Decompressor::inflate(OutputStream *out) {
41 auto stm = (z_stream *) stream;
42 // 输出 buffer 未能完全输出，继续输出
43 if (length != 0) {
44 auto last = length - read;
45 auto real_wrote = out->write(buffer + read, last);
46 if (last != real_wrote) {
47 // 输出 buffer 未能全部输出
48 read += real_wrote;
49 return Z_BUF_ERROR;
50 } else {
51 length = 0;
52 read = 0;
53
54 stm->avail_out = (unsigned int) cap;
55 stm->next_out = buffer;
56 }
57 }
58
59 // 输入 buffer 未解析完全，继续解析
60 int ret;
61 do {
62 ret = ::inflate(stm, Z_FINISH);
63 if (ret == Z_STREAM_ERROR)
64 return Z_STREAM_ERROR;
65
66 auto wrote = (int) cap - stm->avail_out;
67 auto real_wrote = out->write(buffer, wrote);
68 if (wrote != real_wrote) {
69 // 输出 buffer 未能全部输出
70 length = wrote;
71 read = real_wrote;
72 return Z_BUF_ERROR;
73 } else {
74 stm->avail_out = (unsigned int) cap;
75 stm->next_out = buffer;
76 }
77 } while (ret != Z_STREAM_END);
78
79 return Z_OK;
