9Value Yaml::parseObject(sese::Yaml::TokensQueue &tokens_queue, size_t level) {
10 if (level == 0) return {};
11
12 auto result = Value::Dict();
13 // 对象无子元素
14 if (tokens_queue.empty()) return Value(std::move(result));
15 int count = std::get<0>(tokens_queue.front());
16
17 while (!tokens_queue.empty()) {
18 decltype(auto) count_and_tokens = tokens_queue.front();
19 auto current_count = std::get<0>(count_and_tokens);
20 auto current_tokens = std::get<1>(count_and_tokens);
21
22 if (count == current_count) {
23 if (current_tokens.size() == 3) {
24 if (current_tokens[1] != ":") {
25 return {};
26 }
27 // 普通键值对
28 std::string key = current_tokens[0];
29 std::string value = current_tokens[2];
30 if (sese::strcmpDoNotCase(value.c_str(), "null") || value == "~") {
31 result.set(key, Value());
32 } else {
33 result.set(key, std::move(value));
34 }
35 tokens_queue.pop();
36 } else if (current_tokens.size() == 2) {
37 // 可能是空对象，也可能是新对象或新数组
38 auto current_count_and_tokens = tokens_queue.front();
39 tokens_queue.pop();
40 if (tokens_queue.empty()) {
41 // 不存在下一行且当前元素为 null
42 result.set(current_tokens[0], Value());
43 } else {
44 // 存在下一行
45 auto next_count_and_tokens = tokens_queue.front();
46 auto next_count = std::get<0>(next_count_and_tokens);
47 auto next_tokens = std::get<1>(next_count_and_tokens);
48 if (next_count == count) {
49 // 当前行元素是 null
50 // 下一个元素依然是本对象所属
51 result.set(current_tokens[0], Value());
52 } else if (next_count < count) {
53 // 当前行元素是 null
54 // 下一个元素不是本对象所属
55 result.set(current_tokens[0], Value());
56 return Value(std::move(result));
57 } else if (next_count > count) {
58 // 新的数组
59 if (next_tokens[0] == "-") {
60 if (auto value_object = parseArray(tokens_queue, level - 1); value_object.isList()) {
61 result.set(current_tokens[0], std::move(value_object));
62 } else {
63 return {};
64 }
65 }
66 // 新的对象
67 else {
68 if (auto value_object = parseObject(tokens_queue, level - 1); value_object.isDict()) {
69 result.set(current_tokens[0], std::move(value_object));
70 } else {
71 return {};
72 }
73 }
74 }
75 }
76 }
77 } else if (count > current_count) {
78 // 当前对象结束
79 return Value(std::move(result));
80 }
81 // 错误 - 这几乎不可能发生
82 // GCOVR_EXCL_START
83 else {
84 tokens_queue.pop();
85 continue;
86 }
87 // GCOVR_EXCL_STOP
88 }
89 return Value(std::move(result));
90}

172Value Yaml::parse(sese::io::InputStream *input, size_t level) {
173 TokensQueue tokens_queue;
174 while (true) {
175 decltype(auto) line = sese::yaml::YamlUtil::getLine(input);
176 if (std::get<1>(line).empty()) {
177 break;
178 }
179 auto tokens = sese::yaml::YamlUtil::tokenizer(std::get<1>(line));
180 tokens_queue.emplace(std::get<0>(line), tokens);
181 }
182
183 decltype(auto) top = tokens_queue.front();
184 decltype(auto) top_tokens = std::get<1>(top);
185 if (top_tokens.empty()) return {};
186 if (top_tokens[0] == "-") {
187 // 根元素是数组
188 return parseArray(tokens_queue, level);
189 } else {
190 // 根元素是对象
191 return parseObject(tokens_queue, level);
192 }
193}

195void Yaml::streamifyObject(io::OutputStream *output, const Value::Dict &dict, size_t level) {
196 for (const auto &[fst, snd]: dict) {
197 if (snd->getType() == Value::Type::DICT) {
198 if (auto &&sub = snd->getDict(); !sub.empty()) {
199 sese::yaml::YamlUtil::writeSpace(level * 2, output);
200 output->write(fst.c_str(), fst.length());
201 output->write(":\n", 2);
202 streamifyObject(output, sub, level + 1);
203 }
204 } else if (snd->getType() == Value::Type::LIST) {
205 if (auto &&sub = snd->getList(); !sub.empty()) {
206 sese::yaml::YamlUtil::writeSpace(level * 2, output);
207 output->write(fst.c_str(), fst.length());
208 output->write(":\n", 2);
209 streamifyArray(output, sub, level + 1);
210 }
211 } else {
212 const auto SUB = snd;
213 sese::yaml::YamlUtil::writeSpace(level * 2, output);
214 output->write(fst.c_str(), fst.length());
215 output->write(": ", 2);
216 if (SUB->isNull()) {
217 output->write("~", 1);
218 } else if (SUB->isBool()) {
219 const auto DATA = SUB->getBool() ? "true" : "false";
220 output->write(DATA, std::strlen(DATA));
221 } else if (SUB->isInt() || SUB->isDouble()) {
222 auto data = SUB->toString();
223 output->write(data.c_str(), data.length());
224 } else if (SUB->isString()) {
225 auto &&data = SUB->getString();
226 output->write(data.c_str(), data.length());
227 }
228 output->write("\n", 1);
229 }
230 }

233void Yaml::streamifyArray(io::OutputStream *output, const Value::List &list, size_t level) {
234 auto count = 0;
235 for (decltype(auto) item: list) {
236 if (item.getType() == Value::Type::DICT) {
237 if (auto &&sub = item.getDict(); !sub.empty()) {
238 auto name = "element_" + std::to_string(count);
239 count += 1;
240 sese::yaml::YamlUtil::writeSpace(level * 2, output);
241 output->write("- ", 2);
242 output->write(name.c_str(), name.length());
243 output->write(":\n", 2);
244 streamifyObject(output, sub, level + 1);
245 }
246 } else if (item.getType() == Value::Type::LIST) {
247 auto &&sub = item.getList();
248 if (!sub.empty()) {
249 auto name = "element_" + std::to_string(count);
250 count += 1;
251 sese::yaml::YamlUtil::writeSpace(level * 2, output);
252 output->write("- ", 2);
253 output->write(name.c_str(), name.length());
254 output->write(":\n", 2);
255 streamifyArray(output, sub, level + 1);
256 }
257 } else {
258 auto &&sub = item;
259 sese::yaml::YamlUtil::writeSpace(level * 2, output);
260 output->write("- ", 2);
261 if (sub.isNull()) {
262 output->write("~", 1);
263 } else if (sub.isBool()) {
264 auto data = sub.getBool() ? "true" : "false";
265 output->write(data, std::strlen(data));

278void Yaml::streamify(io::OutputStream *output, const Value &value) {
279 if (value.getType() == Value::Type::DICT) {
