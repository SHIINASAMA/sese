36template<class RETURN_TYPE>
37std::shared_future<RETURN_TYPE> sese::async(const std::function<RETURN_TYPE()> &task) noexcept {
38 std::packaged_task<RETURN_TYPE()> packaged_task(task);
39 std::shared_future<RETURN_TYPE> future(packaged_task.get_future());
40
41 std::thread(
42 [&](std::packaged_task<RETURN_TYPE()> task) {
43 task();
44 },
45 std::move(packaged_task)
46 )
47 .detach();
48
49 return future;
50}
