55std::optional<std::string> huffman_tree_t::decode(const char *src, size_t len) {
56 std::string dst;
57 huffman_node_t *current(m_root);
58
59 if (len > std::numeric_limits<unsigned int>::max()) {
60 // throw std::invalid_argument("HPACK::huffman_tree_t::decode(): Overly long input string");
61 return std::nullopt;
62 }
63
64 for (unsigned int idx = 0; idx < len; ++idx) {
65 for (int8_t j = 7; j >= 0; j--) {
66 if ((src[idx] & (1 << j)) > 0) {
67 if (nullptr == current->right()) {
68 // throw std::runtime_error(
69 // "HPACK::huffman_tree_t::decode(): Internal state error (right == nullptr)");
70 return std::nullopt;
71 }
72 current = current->right();
73 } else {
74 if (nullptr == current->left()) {
75 // throw std::runtime_error("HPACK::huffman_tree_t::decode(): Internal state error (left == nullptr)");
76 return std::nullopt;
77 }
78 current = current->left();
79 }
80
81 if (current->code() >= 0) {
82 uint16_t code = current->code();
83
84 if (257 == code)
85 dst += static_cast<uint8_t>(((code & 0xFF00) >> 8) & 0xFF);
86
87 dst += static_cast<uint8_t>(code & 0xFF);
88 current = m_root;
89 }
90 }
91 }
92
93 return dst;
94}

111std::vector<uint8_t> huffman_encoder_t::encode(std::vector<uint8_t> &src) noexcept {
112 std::vector<uint8_t> ret(0);
113
114 for (auto &byte: src) {
115 bits_t bits = HUFFMAN_TABLE.at(byte);
116
117 for (decltype(auto) bit: bits) {
118 if (write_bit(bit)) {
119 ret.push_back(m_byte);
120 m_byte = 0;
121 m_count = 8;
122 }
123 }
124 }
125
126 // Apparently the remainder unused bits
127 // are to be set to 1, some sources refer
128 // to this as the EOS bit, but the code
129 // for EOS is like 30-bits of 1's so its
130 // clearly not the EOS code.
131 if (8 != m_count && 0 != m_count) {
132 m_byte = (m_byte << (m_count - 1));
133 m_byte |= (0xFF >> (8 - m_count));
134 ret.push_back(m_byte);
135 m_byte = 0;
136 m_count = 8;
137 }
138
139 return ret;
140}
