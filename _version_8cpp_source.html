51std::string Version::toShortString() const {
52 std::vector<uint16_t> vector;
53 vector.reserve(4);
54
55 size_t count = 0;
56 if (revision_) {
57 vector.emplace_back(revision_);
58 count++;
59 }
60
61 if (!patch_ && !count) {
62 } else {
63 vector.emplace_back(patch_);
64 count++;
65 }
66
67 if (!minor_ && !count) {
68 } else {
69 vector.emplace_back(minor_);
70 count++;
71 }
72
73 if (!major_ && !count) {
74 } else {
75 vector.emplace_back(major_);
76 count++;
77 }
78
79 if (count == 0) {
80 vector.emplace_back(0);
81 count += 1;
82 } else {
83 std::reverse(vector.begin(), vector.end());
84 }
85
86 // vector to versioning string
87 std::string version;
88 for (size_t i = 0; i < count; i++) {
89 version += std::to_string(vector[i]);
90 if (i != count - 1) {
91 version += ".";
92 }
93 }
94
95 return version;
96}
