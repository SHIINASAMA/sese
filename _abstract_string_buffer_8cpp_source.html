222void AbstractStringBuffer::trim() noexcept {
223 size_t w = 0;
224 for (size_t i = 0; i < len; i++) {
225 if (isSpace(this->buffer[len - 1 - i])) {
226 this->buffer[len - 1 - i] = '\0';
227 w += 1;
228 } else {
229 break;
230 }
231 }
232 this->len -= w;
233
234 w = 0;
235 for (int i = 0; i < this->len; i++) {
236 if (!isSpace(this->buffer[i])) {
237 w = i;
238 break;
239 }
240 }
241 for (int i = 0; i < len - w; i++) {
242 this->buffer[i] = this->buffer[i + w];
243 }
244 memset(&this->buffer[len - w], 0, w + 1);
245 this->len -= w;
246}
