123template<class SERVICE>
124bool sese::service::UserBalanceLoader::init(std::function<SERVICE *()> creator) noexcept {
125 if (address == nullptr) return false;
126
127 socket = new net::Socket(
128 address->getRawAddress()->sa_family == AF_INET ? net::Socket::Family::IPv4 : net::Socket::Family::IPv6,
129 net::Socket::Type::TCP
130 );
131 if (-1 == socket->getRawSocket()) {
132 return false;
133 }
134
135 if (!socket->setNonblocking()) {
136 goto freeSocket;
137 }
138
139 if (0 != socket->bind(address)) {
140 goto freeSocket;
141 }
142
143 if (0 != socket->listen(32)) {
144 goto freeSocket;
145 }
146
147 masterEventLoop = new MasterEventLoop;
148 masterEventLoop->setListenFd(static_cast<int>(socket->getRawSocket()));
149 if (!masterEventLoop->init()) {
150 goto freeMaster;
151 }
152
153 for (size_t i = 0; i < threads; ++i) {
154 auto event = creator();
155 if (event == nullptr) {
156 goto freeEvent;
157 }
158 if (!event->init()) {
159 delete event;
160 goto freeEvent;
161 } else {
162 eventLoopVector.emplace_back(event);
163 }
164 }
165
166 // 初始化交换队列
167 masterSocketQueueArray = new std::queue<SocketStatus>[threads];
168 slaveSocketQueueArray = new std::queue<SocketStatus>[threads];
169 mutexArray = new std::mutex[threads];
170
171 return true;
172
173freeEvent:
174 for (decltype(auto) event_loop: eventLoopVector) {
175 delete event_loop;
176 }
177 eventLoopVector.clear();
178
179freeMaster:
180 delete masterEventLoop;
181 masterEventLoop = nullptr;
182
183freeSocket:
184 socket->close();
185 delete socket;
186 socket = nullptr;
187 return false;
188}
