124template<class SERVICE>
125bool sese::service::UserBalanceLoader::init(std::function<SERVICE *()> creator) noexcept {
126 if (address == nullptr) return false;
127
128 socket = new net::Socket(
129 address->getRawAddress()->sa_family == AF_INET ? net::Socket::Family::IPv4 : net::Socket::Family::IPv6,
130 net::Socket::Type::TCP
131 );
132 if (-1 == socket->getRawSocket()) {
133 return false;
134 }
135
136 if (!socket->setNonblocking()) {
137 goto freeSocket;
138 }
139
140 if (0 != socket->bind(address)) {
141 goto freeSocket;
142 }
143
144 if (0 != socket->listen(32)) {
145 goto freeSocket;
146 }
147
148 masterEventLoop = new MasterEventLoop;
149 masterEventLoop->setListenFd(static_cast<int>(socket->getRawSocket()));
150 if (!masterEventLoop->init()) {
151 goto freeMaster;
152 }
153
154 for (size_t i = 0; i < threads; ++i) {
155 auto event = creator();
156 if (event == nullptr) {
157 goto freeEvent;
158 }
159 if (!event->init()) {
160 delete event;
161 goto freeEvent;
162 } else {
163 eventLoopVector.emplace_back(event);
164 }
165 }
166
167 // 初始化交换队列
168 masterSocketQueueArray = new std::queue<SocketStatus>[threads];
169 slaveSocketQueueArray = new std::queue<SocketStatus>[threads];
170 mutexArray = new std::mutex[threads];
171
172 return true;
173
174freeEvent:
175 for (decltype(auto) event_loop: eventLoopVector) {
176 delete event_loop;
177 }
178 eventLoopVector.clear();
179
180freeMaster:
181 delete masterEventLoop;
182 masterEventLoop = nullptr;
183
184freeSocket:
185 socket->close();
186 delete socket;
187 socket = nullptr;
188 return false;
189}
