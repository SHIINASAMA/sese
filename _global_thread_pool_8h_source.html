46template<class RETURN_TYPE>
47std::shared_future<RETURN_TYPE> sese::GlobalThreadPool::postTask(const std::function<RETURN_TYPE()> &func) {
48 auto task2 = new Task2<RETURN_TYPE>();
49 auto packaged_task = std::packaged_task<RETURN_TYPE()>(func);
50 std::shared_future<RETURN_TYPE> future(packaged_task.get_future());
51 task2->packagedTask = std::move(packaged_task);
52 QueueUserWorkItem(taskRunner2<RETURN_TYPE>, task2, WT_EXECUTEDEFAULT);
53 return future;
54}
