279template<>
280struct text::overload::Formatter<Value> {
281 size_t level = 4;
282
283 bool parse(const std::string &args) {
284 char *end;
285 level = static_cast<size_t>(std::strtol(args.c_str(), &end, 10));
286 if (level == 0) {
287 return false;
288 }
289 return true;
290 }
291
292 void format(FmtCtx &ctx, Value &value) const {
293 value.toString(ctx.builder, level);
294 }
