134bool impl::SqlitePreparedStatementImpl::setText(uint32_t index, const char *value) noexcept {
135 if (this->stmtStatus) sqlite3_reset(stmt);
136 if (this->isManual[index - 1]) {
137 free(this->buffer[index - 1]);
138 }
139
140 this->isManual[index - 1] = true;
141 auto len = strlen(value);
142 this->buffer[index - 1] = malloc(len + 1);
143 static_cast<char *>(this->buffer[index - 1])[len] = 0;
144 memcpy(this->buffer[index - 1], value, len);
145
146 this->stmtStatus = false;
147 return SQLITE_OK == sqlite3_bind_text(stmt, static_cast<int>(index), value, -1, nullptr);
148}

195bool impl::SqlitePreparedStatementImpl::getColumnType(uint32_t index, MetadataType &type) noexcept {
196 if (sqlite3_column_count(stmt) <= static_cast<int>(index)) {
197 return false;
198 }
199
200 auto raw = sqlite3_column_decltype(stmt, static_cast<int>(index));
201 if (nullptr == raw) {
202 return false;
203 }
204
205 // 这并不是最好的确定 SQLite 结果集元数据的最好的方法，
206 // 但目前的官方提供的 API 使得我暂时只能这么做
207 auto target = splitBefore(raw);
208 // SESE_DEBUG("raw: %s, target: %s", raw, target.c_str());
209 if (in(INTEGER_AFFINITY_SET, 9, target.c_str())) {
210 type = MetadataType::INTEGER;
211 } else if (in(TEXT_AFFINITY_SET, 8, target.c_str())) {
212 type = MetadataType::TEXT;
213 } else if (in(REAL_AFFINITY_SET, 4, target.c_str())) {
214 type = MetadataType::FLOAT;
215 } else if (sese::strcmpDoNotCase("BOOLEAN", target.c_str())) {
216 type = MetadataType::BOOLEAN;
217 } else if (sese::strcmpDoNotCase("DATE", target.c_str())) {
218 type = MetadataType::DATE;
219 } else if (sese::strcmpDoNotCase("DATETIME", target.c_str())) {
220 type = MetadataType::DATE_TIME;
221 } else {
222 type = MetadataType::UNKNOWN;
223 }
224
225 return true;
226}

228int64_t impl::SqlitePreparedStatementImpl::getColumnSize(uint32_t index) noexcept {
229 if (sqlite3_column_count(stmt) <= static_cast<int>(index)) {
230 return -1;
231 }
232
233 std::string raw = sqlite3_column_decltype(stmt, static_cast<int>(index));
234 if (raw.empty()) {
235 return -1;
236 }
237
238 // 同上
239 auto pos = raw.find_first_of('(');
240 if (pos == std::string::npos) {
241 return 0;
242 } else {
243 auto number = raw.substr(pos + 1, raw.length() - pos - 2);
244 // SESE_DEBUG("number string: %s", number.c_str());
245 char *end;
246 return std::strtol(number.c_str(), &end, 10);
247 }
