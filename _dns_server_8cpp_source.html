71void sese::net::dns::DnsServer::loop() noexcept {
72 auto client_address = std::make_shared<sese::net::IPv4Address>();
73 while (true) {
74 if (isShutdown) {
75 break;
76 }
77
78 uint8_t buffer[DNS_PACKAGE_SIZE];
79 auto len = socket->recv(buffer, sizeof(buffer), client_address, 0);
80 if (len <= 0) {
81 sese::sleep(0);
82 continue;
83 }
84
85 auto input = sese::io::InputBufferWrapper((const char *) buffer + 12, len - 12);
86 auto output = sese::io::OutputBufferWrapper((char *) buffer + 12, sizeof(buffer) - 12);
87
88 FrameHeaderInfo info;
89 DnsSession session;
90 DnsUtil::decodeFrameHeaderInfo(buffer, info);
91 DnsUtil::decodeQueries(info.questions, &input, session.getQueries());
92
93 info.flags.rcode = SESE_DNS_RCODE_NO_ERROR;
94 info.flags.RD = 0;
95 info.flags.QR = 1;
96 for (auto &q: session.getQueries()) {
97 if (q.getType() == SESE_DNS_QR_TYPE_A) {
98 auto iterator = hostIPv4Map.find(q.getName());
99 if (iterator == hostIPv4Map.end()) {
100 info.flags.rcode = SESE_DNS_RCODE_NAME_ERROR;
101 continue;
102 } else {
103 uint16_t offset = q.getOffset();
104 offset |= 0b1100'0000'0000'0000;
105 offset = ToBigEndian16(offset);
106 std::string name = {(const char *) &offset, 2};
107 session.getAnswers().emplace_back(
108 name,
109 SESE_DNS_QR_TYPE_A,
110 SESE_DNS_QR_CLASS_IN,
111 600,
112 4,
113 iterator->second
114 );
115 }
116 info.answerPrs += 1;
117 } else if (q.getType() == SESE_DNS_QR_TYPE_AAAA) {
118 auto iterator = hostIPv6Map.find(q.getName());
119 if (iterator == hostIPv6Map.end()) {
120 info.flags.rcode = SESE_DNS_RCODE_NAME_ERROR;
121 continue;
122 } else {
123 uint16_t offset = q.getOffset();
124 offset |= 0b1100'0000'0000'0000;
125 offset = ToBigEndian16(offset);
126 std::string name = {(const char *) &offset, 2};
127 session.getAnswers().emplace_back(
128 name,
129 SESE_DNS_QR_TYPE_AAAA,
130 SESE_DNS_QR_CLASS_IN,
131 600,
132 16,
133 iterator->second
134 );
135 }
136 info.answerPrs += 1;
137 } else {
138 info.flags.rcode = SESE_DNS_RCODE_NOT_IMPLEMENTED;
139 continue;
140 }
141 }
142
143 DnsUtil::encodeFrameHeaderInfo(buffer, info);
144 DnsUtil::encodeQueries(&output, session.getQueries());
145 DnsUtil::encodeAnswers(&output, session.getAnswers());
146 socket->send(buffer, 12 + output.getLength(), client_address, 0);
147 }
