56void Timer::loop() noexcept {
57 while (!isShutdown) {
58 size_t index = currentTimestamp % number;
59 mutex.lock();
60 for (auto iterator = timerTasks[index].begin(); iterator != timerTasks[index].end();) {
61 TimerTask::Ptr task = *iterator;

63 if (currentTimestamp == task->targetTimestamp) { // GCOVR_EXCL_LINE
64 task->callback();
65 iterator = timerTasks[index].erase(iterator);
